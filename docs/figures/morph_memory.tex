\node[font=\normalsize\sffamily, text=bdr, align=left, anchor=north west]
  at ($(explain.north west)+(0.4,-0.85)$) {%
  \textbf{1.}\; Backbone processes a \textbf{chunk} of tokens through
    prelude $\to$ core $\to$ coda\\[4pt]
  \textbf{2.}\; Hidden states $h$ are projected:\;
    $k = W_k(h)$\; {\scriptsize (address)},\;
    $v = W_v(h)$\; {\scriptsize (content to store)}\\[4pt]
  \textbf{3.}\; Memory queries itself:\;
    $M(k)$ = ``what do I already know at this address?''\\[4pt]
  \textbf{4.}\; \textbf{Surprise} $= \|M(k) - v\|^2$\;
    {\scriptsize --- high if $v$ is new, low if already stored}\\[4pt]
  \textbf{5.}\; Gradient descent on $M$'s weights:\;
    surprised $\to$ large update,\;
    unsurprised $\to$ small update};
